Table customer {
  customer_id int [pk]
  username varchar
  email varchar
  password varchar
  created_at timestamp
  updated_at timestamp
}

Table product {
  product_id int [pk]
  name varchar
  description text
  price decimal
  category varchar
  age_range varchar
  availability bool
  created_at timestamp
  updated_at timestamp
}

Table cart {
  cart_id int [pk]
  customer_id int
  product_id int
  quantity int
  added_at timestamp
}

Table order {
  customer_id int
  total_amount decimal
  status varchar
  order_date timestamp
  payment_method_id int
  order_id int [pk]
}

Table order_detail {
  order_detail_id int [pk]
  order_id int
  customer_id int
  product_id int
  quantity int
  unit_price decimal
}

Table payment_method {
  customer_id int
  name varchar
  payment_method_id int [pk]
  saved bool
}

Table wishlist {
  wishlist_id int [pk]
  customer_id int
  event_name varchar
  event_date date
  product_ids int[]  //Updated field to store product IDs directly
  created_at timestamp
  updated_at timestamp
}

Table employee {
  employee_id int [pk]
  order_id int
  username varchar
  email varchar
  password varchar
  role varchar
  created_at timestamp
  updated_at timestamp
}

Table admin {
  admin_id int [pk]
  username varchar
  email varchar
  password varchar
  created_at timestamp
  updated_at timestamp
}

Ref: cart.customer_id > customer.customer_id
Ref: cart.product_id > product.product_id
Ref: order.customer_id > customer.customer_id
Ref: order_detail.order_id > order.order_id
Ref: order_detail.product_id > product.product_id
Ref: payment_method.customer_id > customer.customer_id
Ref: order.payment_method_id > payment_method.payment_method_id
Ref: order.order_id > employee.order_id
Ref: wishlist.customer_id > customer.customer_id
Ref: order_detail.customer_id > customer.customer_id
